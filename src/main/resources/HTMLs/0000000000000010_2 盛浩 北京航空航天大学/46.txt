    //初始化
    $(function () {
        var request = g_getUrlParms();
        var searchType = request["type"];
        var artType = request["artType"];
        if (searchType != undefined || searchType != '') {
            $('#searchDiv>a').removeClass('current');
            switchBindSearchType(searchType);
        }
        if (artType && artType != "Articles") {
            $("#articleSearchType").val(artType);
            var text = $("#articleSearchType").find("option[value='" + artType + "']").html();
            $(".select-items").find("li:selected").removeClass("selected");
            $(".select-items").find("li:contains('" + text + "')").addClass("selected");

        }

            $('#searchDiv>a').attr('class', '');
            $('#searchType').val($(this).attr('searchType'));
            $(this).addClass('current');
            var searchType = $('#searchType').val();
            if (searchType == "article") {
                $("#seniorSearchBox").show();
            }
            else if (searchType == "website") {
                $("#seniorSearchBox").hide();
            }
            switchBindSearchType(searchType);

        });

    function switchBindSearchType(searchType) {
        switch (searchType) {
            case "article":
                $('#searchDiv>a[searchType="article"]').addClass('short current');
                $('#searchType').val('article');
                $('#selSearType').show();

                $('#searchType').val('article');
                break;
        }
    }

    function bindSearchKeys() {
        var request = g_getUrlParms();
        var searchKeys = request["key"];
        if (searchKeys) {
            var uvalue = searchKeys.match(/U=[^\[]*/i);
            if (uvalue) {
                var value = uvalue[0].split('=');

                $("#txtMainSearchType").val(searchvalue); //检索框赋值
            } else {
                
            }
        } else {
            
        }

    //自动补全方法
    function bindAutoComplete() {
        var searchType = $('#searchType').val();
        if (searchType == "website")
            return;
        $("#txtMainSearchType").AutoComplete({
            'data': "/ajax/AutoCompleteHandler.ashx",
            "ajaxParams": function () {
                var type = "articles";
                if ($("#hifOutObjectType").val()) {
                    type = $("#hifOutObjectType").val();
                }
                return { "objtype": type.toLowerCase() };
            },
            'ajaxDataType': 'json',
            'width': 'auto', //490

            'onerror': function (msg) { alert(msg); },
            "afterSelectedHandler": function () {
                search();
            }
        });
    }

    //简单检索表达式
    function search() {
        var searType = $('#searchType').val();
        var type = "";
        type = $('#articleSearchType option').eq($('.select-items > ul > li[class="selected"]').index()).val();
        var value = $("#txtMainSearchType").val().replace(/(\;|\,|\<|\>)/g, "").replace(/(\))/g, "）").replace(/(\()/g, "（").replace(/(\[)/g, "0x005b").replace(/(\])/g, "0x005d").replace(/(\')/g, "‘").replace(/(\')/g, "’").replace(/(\")/g, "“").replace(/(\")/g, "”").replace(/(\=)/g, "0x005a");
        var key = "U";
        if (value == "" || $("#txtMainSearchType").hasClass('watermark')) {
            alert("请输入检索条件");
            return false;
        } else {
            if (searType == 'article') {
                geturl(type, key + "=" + value);
            }
            else {
                location.href = '/websearch.aspx?type=website&keywords=' + value;
            }
        }
    }
    function geturl(type, str, searType, rf, index) {
        var url = "", key = "";
        if (str) {
            str = encodeURIComponent(str);
            key = "?key=" + str;
        }
        if (type && type != "-1") {
            artType = encodeURIComponent(rf);
            if (key.indexOf("?") != -1) {
                key += "&artType=" + type;
            } else {
                key += "?artType=" + type;
            }
        }
        if (rf) {
            rf = encodeURIComponent(rf);
            if (key.indexOf("?") != -1) {
                key += "&rf=" + rf;
            } else {
                key += "?rf=" + rf;
            }
        }
        if (index) {
            if (key.indexOf("?") != -1) {
                key += "&uf=" + index;
            } else {
                key += "?uf=" + index;
            }
        }
        if (key.length > 1000) {
            alert("检索表达式位数超出");
            return false;
        }
        url = "/zk/search.aspx?";
        switch (searType) {
            case "Writer":
                url = "/writer/writersearch.aspx";
                break; //人物
            case "Organ":
                url = "/organ/organsearch.aspx";
                break; //机构
            case "Subject":
                url = "/subject/subjectsearch.aspx";
                break; //主题
            case "Journal":
                url = "/journal/journalsearch.aspx";
                break; //期刊
            case "Fund":
                url = "/fund/fundsearch.aspx";
                break; //资助
            case "Domain":
                url = "/domain/domainsearch.aspx";
                break; //学科
            case "Area":
                url = "/area/areasearch.aspx";
                break; //地区
            case "Articles":
                url = "/zk/search.aspx";
                break; //文献
            default:
                url = "/zk/search.aspx";
                break;
        }
        location.href = url + key;
    }

    //回车检索
    function search_enterKeydown(event) {
        var e = $.event.fix(event);
        if (e.keyCode === 13) {
            search();
        }
    }

    //URL参数获取
    function GetQueryString(name) {
        var reg = new RegExp("(^|&)" + name + "=([^&]*)(&|$)");
        var r = decodeURIComponent(window.location.search.substr(1)).match(reg);
        if (r != null) return unescape(r[2]); return null;
    }
